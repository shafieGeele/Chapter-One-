üîò Tabs


‚Üí Interactive dynamic tabs controlled by function [Uses state for selected tab, array of tab items can be mapped]
Example:


- when building buttons , I will start here : 

import "./tabButtons.css"

function TabButton({children, isSelected, ...props }){

    return (

           <button className={isSelected ? "active" : null} {...props}>{children}</button>

         
    )
}

export default TabButton


- this is how  I will accept :


import { useState } from "react";
import TabButton from "./TabButton";
import { EXAMPLES } from "../../data";
import Section from "./Section";
import Tabs from "./Tabs";


function TabNContent(){


    const [selectTopic , setSelectTopic] = useState( )

    function handleSelect(buttonNames){

        setSelectTopic(buttonNames)
        
    }


    let tabContent = " Please click a button "

    if (selectTopic) {
        
        //call the button names inside a data

        tabContent = 
        <>
            <h3>{EXAMPLES[selectTopic].title}</h3>
            <p>{EXAMPLES[selectTopic].description}</p>
            <pre>
                <code>
                        {EXAMPLES[selectTopic].code}
                </code>
            </pre>
        
        </>
    }


    return ( 

        <>

            <Section id="examples" title="EXAMPLES">

                <Tabs  
                
                buttons={
                    <>
                         
               <TabButton isSelected={selectTopic === "components"} onClick={()=>handleSelect("components")}>COMPOENTS</TabButton>
                <TabButton isSelected={selectTopic === "jsx"} onClick={()=>handleSelect("jsx")}>JSX</TabButton>
                <TabButton isSelected={selectTopic === "props"} onClick={()=>handleSelect("props")}>PROPS</TabButton>
                <TabButton isSelected={selectTopic === "state"} onClick={()=>handleSelect("state")}>STATE</TabButton>                    
                    
                    </>
                } >


                             
                <div id="tab-content">

                     {tabContent}



                </div>
        

                </Tabs>

               
  

            </Section>
                
     
        </>


    )
}


export default TabNContent
